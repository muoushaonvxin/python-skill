	import threading
	import time

	def sayhi(num):
		print("running on number: %s" % num)
		time.sleep(3)


	t_list = []
	for i in range(10):
		t = threading.Thread(target=sayhi, args=[i,])
		t.start()
		t_list.append(t)

	for i in t_list:
		i.join()
		
	print("---main---")
