	点击按钮的时候，是没有任何操作的，必须要指定点击这个按钮的时候要干什么

	1. command
	2. bind

	from tkinter import *
	
	def f1():
		global root
		l = Label(root,text='hello python!',font=('',20),fg='yellow')
		l.pack()

	root = Tk()
	root.geometry('400x400')
	b1 = Button(root,text='打印',command=f1)
	b1.pack(anchor=E)
	mainloop()

	from tkinter import *
	
	def f1():
		global root
		l = Label(root,text='hello python!',font=('',20),fg='yellow')

	root = Tk()
	root.geometry('400x400')
	b1 = Button(root,text='打印')
	b1.bind('<Button-1>',f1)
	b1.pack()
	mainloop()

		l.pack()

	root = Tk()
	root.geometry('400x400')
	b1 = Button(root,text='打印')

	b1['width'] = 20
	b1['height'] = 5
	b1['bg'] = 'red'
	b1.pack()
	mainloop()
