
shelve 模块是一个简单的 k, v 将内存数据通过文件持久化的模块，可以持久化任何pickle 可支持的 python 数据格式

import shelve

d = shelve.open('shelve_test')      # 打开一个文件

class Test(object)：
def __init__(self, n)：
self.n = n

	t = Test(123)
t2 = Test(1234123)
	name = ["alex",  "rain",  "test"] 

	d["test"] = name        #  持久化列表
	d["t1"] = t                 #  持久化类
	d["t2"] = t2

d.close()
